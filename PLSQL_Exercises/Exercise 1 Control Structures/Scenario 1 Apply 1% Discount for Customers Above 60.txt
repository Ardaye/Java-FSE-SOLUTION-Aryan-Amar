SET SERVEROUTPUT ON;

DECLARE
    CURSOR c_customers IS
        SELECT CustomerID, DOB
        FROM Customers;
    
    v_customer_id   Customers.CustomerID%TYPE;
    v_dob           Customers.DOB%TYPE;
    v_age           NUMBER;
BEGIN
    OPEN c_customers;
    LOOP
        FETCH c_customers INTO v_customer_id, v_dob;
        EXIT WHEN c_customers%NOTFOUND;
        
        -- Calculate age in years
        v_age := FLOOR(MONTHS_BETWEEN(SYSDATE, v_dob) / 12);
        
        IF v_age > 60 THEN
            UPDATE Loans
            SET InterestRate = InterestRate - 1
            WHERE CustomerID = v_customer_id;
            
            DBMS_OUTPUT.PUT_LINE('✓ Discount applied to Customer ID: ' || v_customer_id || ' (Age: ' || v_age || ')');
        END IF;
    END LOOP;
    CLOSE c_customers;
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('✅ Scenario 1 completed.');
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        DBMS_OUTPUT.PUT_LINE('❌ Error: ' || SQLERRM);
END;
/
